@dataclass(frozen=True)
class DataGapPeriod:
    start: datetime
    end: datetime
    duration: timedelta = field(init=False)

    def __post_init__(self):
        object.__setattr__(self, "duration", self.end - self.start)

    def __str__(self):
        return f"{self.start} - {self.end}"
